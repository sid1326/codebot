from chatterbot import ChatBot
from chatterbot.trainers import ChatterBotCorpusTrainer

# Create a new ChatBot instance
chatbot = ChatBot('MyBot')

# Create a new trainer for the chatbot
trainer = ChatterBotCorpusTrainer(chatbot)

# Train the chatbot based on the English corpus (which includes a variety of conversations)
trainer.train("chatterbot.corpus.english")

# Get a response to an input statement
while True:
try:
userinput = input("You: ")
if userinput.lower() == 'exit':
break
response = chatbot.getresponse(userinput)
print(f"{chatbot.name}: {response}")
except (KeyboardInterrupt, EOFError, SystemExit):
break

def calculate_tax(income):
    tax_owed = 0
    
    # Tax bracket structure defined as (upper_limit, tax_rate)
    brackets = [
        (10000, 0.0),
        (50000, 0.1),
        (100000, 0.2),
        (float('inf'), 0.3)  # float('inf') represents any income above $100000
    ]
    
    previous_upper_limit = 0
    for upper_limit, tax_rate in brackets:
        if income > upper_limit:
            # If income exceeds the current upper limit, we need to tax the full range of this bracket
            tax_owed += (upper_limit - previous_upper_limit) * tax_rate
        else:
            # If income is less than the current upper limit, only tax the portion from the previous limit to the actual income
            tax_owed += (income - previous_upper_limit) * tax_rate
            break
        
        previous_upper_limit = upper_limit
    
    return tax_owed

# Test the function with some example income values
if __name__ == "__main__":
    print(calculate_tax(8000))    # Expected output: $0 because 8000 is less than $10,000 (0% tax bracket)
    print(calculate_tax(10000))   # Expected output: $0 because 10000 is still in the 0% tax bracket (until $10,000)
    print(calculate_tax(15000))   # Expected output to be (15000 - 10000) * 0.1 = $5000 * 0.1 = $500
    print(calculate_tax(50000))   # Total tax should be (50000 - 10000) * 0.1 = $40000 * 0.1 = $4000
    print(calculate_tax(70000))   # First $10000 is untaxed, $10000 to $50000 taxed at 10% is $4000, $20000 (above $50000, so part is $20000) taxed at 20% is $4000. Total: $4000 + $4000 = $8000
    print(calculate_tax(200000))  # First $10000 untaxed, next $40000 taxed $4000, next $50000 taxed $10000, remaining $100000 taxed $30000. Total: $4000 + $10000 + $30000 = $44000
